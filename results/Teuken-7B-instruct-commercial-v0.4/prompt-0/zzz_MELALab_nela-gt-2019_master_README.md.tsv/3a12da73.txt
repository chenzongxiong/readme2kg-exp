```python
import sqlite3

# Connect to the database
conn = sqlite3.connect('article_metadata.db')

# Create a cursor object
cursor = conn.cursor()

# Execute SQL queries to load the data
cursor.execute("SELECT id, date, source, title, content, author, published, published_utc, collection_utc FROM articles")
cursor.execute("SELECT DISTINCT source FROM articles")

# Close the cursor and connection
cursor.close()
conn.close()
```
